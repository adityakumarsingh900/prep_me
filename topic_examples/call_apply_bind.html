Call vs Apply vs Bind 

   function greet(c1, c2) {
    console.log(`Hello, my name is ${this.name} and I know ${c1} and ${c2}`);
  }
  
  const john = {name: 'John'};
  greet.call(john, 'Python', 'JavaScript');
  // Outputs: "Hello, my name is John and I know Python and JavaScript"

  greet.apply(john, ['Python', 'JavaScript']);
  // Outputs: "Hello, my name is John and I know Python and JavaScript"

  const johnGreet = greet.bind(john, 'Python', 'JavaScript');
  johnGreet();
  // Outputs: "Hello, my name is John and I know Python and JavaScript"

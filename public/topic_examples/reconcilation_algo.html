Same Root Element Type 

    // Old output
    <div className="old">Hello</div>

    // New output
    <div className="new">Hello</div>

    // React will update the class of the existing div instead of creating a new
    one.
Different Root Element Types 

    // Old output
    <div>Hello</div>

    // New output
    <span>Hello</span>

    // React will replace the entire div with a new span.
on Children 

    // Old output
    <ul>
      <li key="1">Item 1</li>
      <li key="2">Item 2</li>
    </ul>

    // New output
    <ul>
      <li key="1">Item 1</li>
      <li key="2">Item 2</li>
      <li key="3">Item 3</li>
    </ul>

    // React will add a new li for Item 3, instead of recreating the entire
    list.
Same Component Type 

    // Old output
    <MyComponent key="1" prop="old" />

    // New output
    <MyComponent key="1" prop="new" />

    // React will keep the MyComponent instance and update its prop.
